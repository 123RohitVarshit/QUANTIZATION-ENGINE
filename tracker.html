#include <stdio.h>
#include <stdint.h>
#include <math.h>     // for roundf()

/* Quantize a single float to int8
   Formula: q = clamp(round(x / scale), -128, 127)  */
int8_t quantize_scalar(float x, float scale) {
    float q_float = roundf(x / scale);

    // Clamp to int8 range [-128, 127]
    if (q_float >  127.0f) q_float =  127.0f;
    if (q_float < -128.0f) q_float = -128.0f;

    return (int8_t)q_float;
}

/* Dequantize int8 back to float */
float dequantize_scalar(int8_t q, float scale) {
    return (float)q * scale;
}

int main() {
    float scale = 0.05f;

    // Test values
    float test[] = { 1.2f, -0.5f, 3.1f, -2.8f, 0.0f };
    int   n      = 5;

    printf("%-10s %-10s %-12s %-10s\n",
           "original", "quantized", "dequantized", "error");
    printf("------------------------------------------\n");

    for (int i = 0; i < n; i++) {
        int8_t q    = quantize_scalar(test[i], scale);
        float  back = dequantize_scalar(q, scale);
        float  err  = test[i] - back;
        printf("%-10.4f %-10d %-12.4f %-10.4f\n",
               test[i], q, back, err);
    }
    return 0;
}

/* Quantize a whole array of floats to int8
   input  : float array  (length n)
   output : int8_t array (length n) — caller must allocate this!
   scale  : quantization scale                               */
void quantize_array(float* input, int8_t* output, int n, float scale) {
    for (int i = 0; i < n; i++) {
        float q = roundf(input[i] / scale);
        if (q >  127.0f) q =  127.0f;
        if (q < -128.0f) q = -128.0f;
        output[i] = (int8_t)q;
    }
}

    int8_t quantized[8];   // output array — same length
    int    n     = 8;
    float  scale = 0.02f;

    quantize_array(weights, quantized, n, scale);

    printf("Weights : ");
    for (int i = 0; i < n; i++) printf("%6.2f ", weights[i]);
    printf("\n");

    printf("Int8    : ");
    for (int i = 0; i < n; i++) printf("%6d ", quantized[i]);
    printf("\n");

    return 0;
}

#include <stdio.h>
#include <stdint.h>
#include <math.h>

/* Compute absmax scale for symmetric int8 quantization
   scale = max(|x|) / 127                               */
float compute_scale_absmax(float* data, int n) {
    float max_abs = 0.0f;
    for (int i = 0; i < n; i++) {
        float abs_val = fabsf(data[i]);
        if (abs_val > max_abs) max_abs = abs_val;
    }
    return max_abs / 127.0f;
}

void quantize_array(float* in, int8_t* out, int n, float scale) {
    for (int i = 0; i < n; i++) {
        float q = roundf(in[i] / scale);
        if (q >  127.0f) q =  127.0f;
        if (q < -128.0f) q = -128.0f;
        out[i] = (int8_t)q;
    }
}

    // Auto-compute scale from data
    float scale = compute_scale_absmax(weights, n);

}

#include <stdio.h>
#include <stdint.h>
#include <math.h>

/* Mean Squared Error between original and dequantized */
float compute_mse(float* original, int8_t* quantized,
                   int n, float scale) {
    float sum = 0.0f;
    for (int i = 0; i < n; i++) {
        float dequant = (float)quantized[i] * scale;
        float diff    = original[i] - dequant;
        sum += diff * diff;
    }
    return sum / (float)n;
}

float compute_scale_absmax(float* data, int n) {
    float max_abs = 0.0f;
    for (int i = 0; i < n; i++) {
        float a = fabsf(data[i]);
        if (a > max_abs) max_abs = a;
    }
    return max_abs / 127.0f;
}

int main() {
    float  data[] = { 0.5f, -1.0f, 1.5f, -0.25f,
                      0.75f, -0.5f, 1.0f, -1.5f };
    int8_t quantized[8];
    int    n     = 8;
    float  scale = compute_scale_absmax(data, n);

    for (int i = 0; i < n; i++) {
        float q = roundf(data[i] / scale);
        if (q >  127) q =  127;
        if (q < -128) q = -128;
        quantized[i] = (int8_t)q;
    }

    float mse = compute_mse(data, quantized, n, scale);
    printf("Scale : %.6f\n", scale);
    printf("MSE   : %.8f\n", mse);
    printf("RMSE  : %.8f\n", sqrtf(mse));

    return 0;
}

#include <stdio.h>
#include <stdint.h>
#include <math.h>

/* QuantParams: same concept as torch's per-tensor qparams */
typedef struct {
    float   scale;
    int32_t zero_point;   // 0 for symmetric, nonzero for asymmetric
} QuantParams;

/* Calibrate params from data using absmax (symmetric) */
QuantParams calibrate_absmax(float* data, int n) {
    float max_abs = 0.0f;
    for (int i = 0; i < n; i++) {
        float a = fabsf(data[i]);
        if (a > max_abs) max_abs = a;
    }
    QuantParams p;
    p.scale      = max_abs / 127.0f;
    p.zero_point = 0;
    return p;
}

int8_t quantize(float x, QuantParams p) {
    float q = roundf(x / p.scale) + (float)p.zero_point;
    if (q >  127) q =  127;
    if (q < -128) q = -128;
    return (int8_t)q;
}

float dequantize(int8_t q, QuantParams p) {
    return ((float)q - (float)p.zero_point) * p.scale;
}

int main() {
    float w[] = { 0.3f, -0.9f, 1.1f, -0.4f, 0.7f };
    int   n   = 5;

    // Calibrate — get params automatically
    QuantParams p = calibrate_absmax(w, n);
    printf("scale=%.6f  zero_point=%d\n", p.scale, p.zero_point);

    // Quantize and dequantize
    for (int i = 0; i < n; i++) {
        int8_t q    = quantize(w[i], p);
        float  back = dequantize(q, p);
        printf("%.3f → %4d → %.3f\n", w[i], q, back);
    }
    return 0;
}

    zero_point: int = 0

def calibrate_absmax(data):

sym_vs_asym.c
python equivalent
#include <stdio.h>
#include <stdint.h>
#include <math.h>

typedef struct { float scale; int32_t zero_point; } QuantParams;

/* Symmetric: zero_point = 0, scale = max(|x|)/127 */
QuantParams calibrate_symmetric(float* data, int n) {
    float max_abs = 0.0f;
    for (int i = 0; i < n; i++) {
        float a = fabsf(data[i]);
        if (a > max_abs) max_abs = a;
    }
    QuantParams p = { max_abs / 127.0f, 0 };
    return p;
}

/* Asymmetric: uses full uint8 range [0,255]
   scale = (max - min) / 255
   zp    = round(-min / scale)                */
QuantParams calibrate_asymmetric(float* data, int n) {
    float min_val = data[0], max_val = data[0];
    for (int i = 1; i < n; i++) {
        if (data[i] < min_val) min_val = data[i];
        if (data[i] > max_val) max_val = data[i];
    }
    float   scale = (max_val - min_val) / 255.0f;
    int32_t zp    = (int32_t)roundf(-min_val / scale);
    QuantParams p  = { scale, zp };
    return p;
}

int main() {
    // ReLU-like output — all positive, asymmetric is better here
    float activations[] = { 0.0f, 0.1f, 0.8f, 1.2f,
                            0.3f, 2.5f, 0.0f, 1.7f };
    int n = 8;

    QuantParams sym  = calibrate_symmetric(activations, n);
    QuantParams asym = calibrate_asymmetric(activations, n);

    printf("Symmetric : scale=%.4f  zero_point=%d\n",
           sym.scale, sym.zero_point);
    printf("Asymmetric: scale=%.4f  zero_point=%d\n",
           asym.scale, asym.zero_point);

    return 0;
}

acts = np.array([0.0, 0.1, 0.8, 1.2,

#include <stdio.h>
#include <stdint.h>
#include <math.h>

typedef struct {
    float   scale;
    int32_t zero_point;
} QuantParams;

QuantParams calibrate(float* data, int n) {
    float max_abs = 0.0f;
    for (int i = 0; i < n; i++) {
        float a = fabsf(data[i]);
        if (a > max_abs) max_abs = a;
    }
    QuantParams p = { max_abs / 127.0f, 0 };
    return p;
}

void quantize_tensor(float* in, int8_t* out, int n, QuantParams p) {
    for (int i = 0; i < n; i++) {
        float q = roundf(in[i] / p.scale) + p.zero_point;
        if (q >  127) q =  127;
        if (q < -128) q = -128;
        out[i] = (int8_t)q;
    }
}

void dequantize_tensor(int8_t* in, float* out, int n, QuantParams p) {
    for (int i = 0; i < n; i++)
        out[i] = ((float)in[i] - p.zero_point) * p.scale;
}

float mse(float* a, float* b, int n) {
    float s = 0.0f;
    for (int i = 0; i < n; i++) { float d = a[i]-b[i]; s += d*d; }
    return s / n;
}

int main() {
    float weights[] = {
        0.15f, -0.92f, 0.44f,  1.23f,
       -0.34f,  0.68f, -1.10f, 0.05f,
        0.87f, -0.21f,  0.63f, -0.77f
    };
    int    n = 12;
    int8_t quantized[12];
    float  reconstructed[12];

    // Full pipeline
    QuantParams p = calibrate(weights, n);
    quantize_tensor(weights, quantized, n, p);
    dequantize_tensor(quantized, reconstructed, n, p);

    printf("=== Mini Quantization Engine ===\n");
    printf("scale=%.6f  zero_point=%d\n\n", p.scale, p.zero_point);
    printf("%-12s %-8s %-12s %-10s\n",
           "original", "int8", "reconstructed", "error");
    printf("------------------------------------------\n");

    for (int i = 0; i < n; i++) {
        printf("%-12.4f %-8d %-12.4f %-10.4f\n",
               weights[i], quantized[i],
               reconstructed[i], weights[i] - reconstructed[i]);
    }

    printf("\nMSE  : %.8f\n", mse(weights, reconstructed, n));
    printf("RMSE : %.8f\n", sqrtf(mse(weights, reconstructed, n)));

    return 0;
}
